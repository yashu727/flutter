import 'package:flutter/material.dart';

class ButtonToggleScreen extends StatefulWidget {
  const ButtonToggleScreen({super.key});

  @override
  State<ButtonToggleScreen> createState() => _ButtonToggleScreenState();
}

class _ButtonToggleScreenState extends State<ButtonToggleScreen> {
  bool showImage = false;
  bool showDetails = false;

  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Button Toggle Example")),
      body: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          // Buttons Row
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              ElevatedButton(
                onPressed: () {
                  setState(() {
                    showImage = true;
                    showDetails = false;
                  });
                },
                child: const Text("A: Show Image"),
              ),
              const SizedBox(width: 20),
              ElevatedButton(
                onPressed: () {
                  setState(() {
                    showImage = false;
                    showDetails = true;
                  });
                },

              ),
            ],
          ),
          const SizedBox(height: 40),
